## 三、敌人行为模式

### 3.1 chase（追逐）

最基础的 AI 行为：直接向玩家移动，近距离开启碰撞伤害。

```javascript
if (dist > 5) {
    const speed = e.speed * e.speedMult * dt;
    e.x += (dx / dist) * speed + e.knockbackX * dt;
    e.y += (dy / dist) * speed + e.knockbackY * dt;
}
// 碰撞伤害
attackTimer -= dt;
if (attackTimer <= 0 && dist < e.radius + player.radius + 5) {
    PlayerSystem.takeDamage(e.damage);
    attackTimer = e.attackCooldown;
}
```

### 3.2 ranged（远程射击）

保持偏好距离，向玩家发射直线子弹。

```javascript
if (dist < preferredDist - 30) {  // 太近 → 后退
    e.x -= (dx / dist) * speed;
    e.y -= (dy / dist) * speed;
} else if (dist > preferredDist + 30) {  // 太远 → 靠近
    e.x += (dx / dist) * speed;
    e.y += (dy / dist) * speed;
}
// 远程攻击
if (attackTimer <= 0 && dist < 500) {
    BulletSystem.create(e.x, e.y, angle, e.damage, bulletSpeed, 0, false);
    attackTimer = e.attackCooldown;
}
```

### 3.3 explode（自爆）— 新增

快速冲向玩家，死亡时对周围造成范围伤害。

```javascript
// 追逐玩家（比 chase 更快的加速）
if (dist > 5) {
    const chargeSpeed = e.speed * (1 + 0.3 * Math.min(1, 300 / Math.max(1, dist)));
    e.x += (dx / dist) * chargeSpeed * dt;
    e.y += (dy / dist) * chargeSpeed * dt;
}
// 碰撞伤害（触碰即爆炸）
if (dist < e.radius + player.radius + 5) {
    // 自爆：对周围所有敌人和玩家造成范围伤害
    const explosionRadius = 80;
    for (const other of enemies) {
        if (!other.alive || other === e) continue;
        const ex = other.x - e.x, ey = other.y - e.y;
        if (Math.sqrt(ex*ex + ey*ey) < explosionRadius) {
            const dmg = Math.floor(e.damage * 0.5);
            EnemySystem.takeDamage(other, dmg);
        }
    }
    PlayerSystem.takeDamage(e.damage * 1.5);
    ParticleSystem.explosion(e.x, e.y, '#ff4400', 20);
    e.alive = false;
}
// 死亡时爆炸
if (!e.alive && !e._exploded) {
    e._exploded = true;
    // 同上爆炸逻辑
}
```

### 3.4 heal（治疗）— 新增

保持距离，定期治疗附近受伤的友军。

```javascript
// 远离玩家（同 ranged 的距离保持逻辑）
if (dist < preferredDist) {
    // 向后移动
    e.x -= (dx / dist) * speed * dt;
    e.y -= (dy / dist) * speed * dt;
}
// 治疗附近友军
healTimer -= dt;
if (healTimer <= 0) {
    const healRadius = 120;
    for (const other of enemies) {
        if (!other.alive || other === e || other.hp >= other.maxHp) continue;
        const hx = other.x - e.x, hy = other.y - e.y;
        if (Math.sqrt(hx*hx + hy*hy) < healRadius) {
            const healAmt = e.damage * 2;
            other.hp = Math.min(other.maxHp, other.hp + healAmt);
            // 治疗粒子特效
            ParticleSystem.emit(other.x, other.y, 3, {
                speed: 30, color: '#44ff88', life: 0.3, size: 4, type: 'glow'
            });
        }
    }
    healTimer = e.healCooldown || 3.0;
}
```

### 3.5 mortar（迫击）— 新增

缓慢移动，向玩家发射抛物线弹丸（延迟落地，落地后范围伤害）。

```javascript
// 保持极远距离
if (dist < preferredDist) {
    // 远离玩家
    e.x -= (dx / dist) * speed * dt;
    e.y -= (dy / dist) * speed * dt;
}
// 发射迫击弹
mortarTimer -= dt;
if (mortarTimer <= 0 && dist < 600) {
    // 计算落点（玩家当前位置前方预测）
    const predictDist = dist * 0.4;
    const targetX = player.x + player.vx * predictDist;
    const targetY = player.y + player.vy * predictDist;
    // 创建迫击弹（延迟爆炸子弹）
    BulletSystem.createMortar(e.x, e.y, targetX, targetY, e.damage);
    mortarTimer = e.mortarCooldown || 3.0;
    // 发射提示粒子
    ParticleSystem.emit(e.x, e.y, 3, {
        speed: 20, color: '#aa44ff', life: 0.3, size: 3, type: 'glow'
    });
}
```

### 3.6 blink（闪现）— 新增

周期性瞬移到玩家附近攻击，具有闪避能力。

```javascript
blinkTimer -= dt;
if (blinkTimer <= 0 && dist > 100) {
    // 瞬移到玩家附近
    const blinkDist = 80 + Math.random() * 40;
    const blinkAngle = Math.atan2(dy, dx) + (Math.random() - 0.5) * 0.5;
    e.x = player.x + Math.cos(blinkAngle) * blinkDist;
    e.y = player.y + Math.sin(blinkAngle) * blinkDist;
    // 闪现粒子
    ParticleSystem.emit(e.x, e.y, 8, {
        speed: 80, color: '#ff44ff', life: 0.3, size: 5, type: 'glow'
    });
    blinkTimer = e.blinkCooldown || 2.0;
}
// 接近玩家时攻击
if (dist < e.radius + player.radius + 10) {
    // 攻击后立即闪现走
    PlayerSystem.takeDamage(e.damage);
    // 闪现撤退
    const retreatAngle = Math.atan2(dy, dx) + Math.PI + (Math.random() - 0.5) * 0.3;
    e.x = e.x + Math.cos(retreatAngle) * 100;
    e.y = e.y + Math.sin(retreatAngle) * 100;
    blinkTimer = e.blinkCooldown || 2.0;
    ParticleSystem.emit(e.x, e.y, 8, {
        speed: 80, color: '#ff44ff', life: 0.3, size: 5, type: 'glow'
    });
}
// 闪避：受击时概率瞬移
if (刚受击 && Math.random() < 0.3) {
    // 随机瞬移到附近
    e.x += (Math.random() - 0.5) * 200;
    e.y += (Math.random() - 0.5) * 200;
    ParticleSystem.emit(e.x, e.y, 6, { ...闪现粒子 });
}
```

## 五、敌人图标规范

### 5.1 静态图标（256x256）

- 路径: `assets/enemies/cb_enemy_{id}_00001_.png`
- 分辨率: **256x256**
- 风格: 写实风格 PBR 渲染
- 背景: 纯绿色 `#00FF00`（运行时代码自动抠除为透明）
- 构图: 单件孤立，从上方45°俯视角度，正方向朝向画布上方

### 5.2 Sprite 帧动画（行走4方向×4帧）

- 路径: `assets/sprites/enemies/{id}/down-{1~4}.png` 等
- 分辨率: 4x4 grid sprite sheet → agent-sprite-forge 分割为独立帧
- 详情见: `batch_generate_sprites.py`

## 六、生成规则总结

### 6.1 同时存在限制

| 类型 | 限制 |
|------|------|
| `elite` | 最多 1 个同时在场 |
| `boss` | 最多 1 个同时在场 |
| 其他 | 无上限（受 maxEnemies 限制） |

### 6.2 难度缩放

```javascript
difficultyScale = 1 + (currentLevel - 1) * 0.15
spawnRate = max(0.4, 1.5 - (currentLevel - 1) * 0.04)
enemyCount = 4 + floor(currentLevel * 1.5)
```

### 6.3 Boss 生成

```javascript
// 每5关（5/10/15/20...）:
//   第4秒生成一个精英（5~10关）或 Boss（15关以上）
if (currentLevel % 5 === 0 && waveTimer > 4 && !_bossSpawned) {
    bossType = currentLevel >= 15 ? 'boss' : 'elite';
    EnemySystem.create(bossType, x, y, scale * 1.2);
}
```
